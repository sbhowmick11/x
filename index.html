 void reverseArray(int *arr, int m, int n)
 {
    int size;
    size = n;
    int i = 0;
    int j = size - 1;
    while (i < j)
    {
        int t;
        t = arr[i];
        arr[i] = arr[j];
        arr[j] = t;
        i++;
        j--;
    }
 }
