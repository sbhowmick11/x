Problem Statement:
You are given a unsorted array. You must sort the array using quick sort algorithm with an average time complexity of O(nlogn).It is not a stable sorting algprithm.

 C:

#include <stdio.h>
int getindex(int *arr,int low,int high)
{
    int pivot=arr[high];    
    //taking last element as pivot
    int i=low-1;
    for(int j=low;j<high;j++)
    {
        if(arr[j]<pivot)
        {
            i++;
            //swap
            int temp;
            temp=arr[i];
            arr[i]=arr[j];
            arr[j]=temp;
        }
    }
    //swap with pivot
    i++;
    int t;
    t=arr[i];
    arr[i]=arr[high];
    arr[high]=t;
    return i;
}
void quicksort(int *a,int low,int high)
{
    if(low<high)
    {
        int pivot=getindex(a,low,high);
        quicksort(a,low,pivot-1);
        quicksort(a,pivot+1,high);
    }
}
int main()
{
    //printf("Hello World");
    int a[]={6,5,4,30,2,1};int n;
    n=sizeof(a)/sizeof(a[0]);
    quicksort(a,0,n-1);
    for(int i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
    return 0;
}

 C++:

#include <iostream>
#include <vector>
using namespace std;
//Function to swap two elements
void swap(int& a, int& b) 
{
    int temp = a;  a = b;  b = temp;
}
    
//Partition function to divide the array 
//into two subarrays
int partition(vector& arr,int low,int high) 
{
    int pivot = arr[high];    int i = low - 1;
    for (int j = low; j < high; j++) 
    {
        if (arr[j] < pivot) 
        {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}
//Quicksort function
void quickSort(vector& arr, int low, int high)
{
    if (low < high) 
    {
    //Find the partition index
    int partitionIndex = partition(arr, low, high);
    //Recursively sort the two subarrays
    quickSort(arr, low, partitionIndex - 1);
    quickSort(arr, partitionIndex + 1, high);
    }
}
//Utility function to print the array
void printArray(const vector& arr)
{
    for (int i = 0; i < arr.size(); i++) 
    {
        cout << arr[i] << " ";
    }
    cout << endl;
}
int main()
{
    vector arr = {5, 3, 8, 1, 2, 7};
    int n = arr.size();
    cout << "Original array: ";
    printArray(arr);  quickSort(arr, 0, n - 1);
    cout << "Sorted array: ";   printArray(arr);
    return 0;
}

 JAVA:

import java.util.Arrays;
public class QuickSort 
{
    //Function to swap two elements
    public static void swap(int[] arr, int i, int j) 
    {
        int temp = arr[i];  arr[i] = arr[j];
        arr[j] = temp;
    }
    //Partition function to divide the array
    //into two subarrays
    public static int partition(int[] arr, int low, int high)
    {
        int pivot = arr[high];  int i = low - 1;
        for (int j = low; j < high; j++)
        {
            if (arr[j] < pivot) 
            {
                i++;  swap(arr, i, j);
            }
        }
        swap(arr, i + 1, high);   return i + 1;
    }
        
    //Quicksort function
    public static void quickSort(int[] arr,int low,int high)
    {
        if (low < high)
        {
            //Find the partition index
            int partitionIndex = partition(arr, low, high);
            //Recursively sort the two subarrays
            quickSort(arr, low, partitionIndex - 1);
            quickSort(arr, partitionIndex + 1, high);
        }
    }
    //Utility function to print the array
    public static void printArray(int[] arr) 
    {
        for (int i = 0; i < arr.length; i++)
        {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
        
    public static void main(String[] args)
    {
        int[] arr = {5, 3, 8, 1, 2, 7};
        System.out.println("Original array: ");
        printArray(arr);
        quickSort(arr, 0, arr.length - 1);
        System.out.println("Sorted array: ");
        printArray(arr);
    }
}
Explanation:
We are explaining the C++ code,the logics are similar:
1. The code begins by including the necessary libraries: iostream for input/output operations and vector for using a dynamic array.
2. The swap function is defined to swap the values of two integers using a temporary variable.
3. The partition function is defined to divide the array into two subarrays based on a chosen pivot element. It takes three parameters: the array arr, the low index low, and the high index high that specify the range of the array to be partitioned.
4. Inside the partition function, the pivot element is chosen as the last element of the subarray (arr[high]). The variable i is initialized to low - 1, which represents the index of the smaller element.
5. A loop is executed from low to high - 1 to compare each element with the pivot element. If an element is smaller than the pivot, i is incremented, and the swap function is called to swap the current element with the element at index i.
6. After the loop, the pivot element is placed in its correct sorted position by swapping it with the element at index i + 1.
7. The partition function returns the index of the pivot element in its sorted position.
8. The quickSort function is defined to recursively sort the subarrays. It takes three parameters: the array arr, the low index low, and the high index high that specify the range of the array to be sorted.
9. Inside the quickSort function, an if condition is checked to ensure that low is less than high. If it is true, the partition index is obtained by calling the partition function.
10. The quickSort function is then called recursively twice to sort the two subarrays: one from low to partitionIndex - 1 and the other from partitionIndex + 1 to high.
11. The printArray function is defined to print the elements of the array. It takes the array arr as a parameter and prints each element separated by a space.
12. In the main function, an example array arr is initialized with values {5, 3, 8, 1, 2, 7}.
13. The original array is printed using the printArray function.
14. The quickSort function is called to sort the array in place.
15. Finally, the sorted array is printed using the printArray function.

The code follows a divide-and-conquer approach by recursively partitioning the array into smaller subarrays and sorting them separately. The choice of the pivot element and its placement in the correct position is crucial for the efficiency of the algorithm. Quicksort has an average-case time complexity of O(n log n) and a worst-case time complexity of O(n^2) when the pivot selection is not optimal.
